## Installation

```shell
$ yarn add styled-components react-analog-clock
```

or

```shell
$ npm install --save styled-components react-analog-clock
```

---

## Usage

```JSX
// code away!

import AnalogClock from 'react-analog-clock';


const clockOptions = {
    baseColor: '#ffffff',
    borderColor: '#000000',
    borderWidth: 5,
    centerColor: '#000000',
    handColors: {
        hour: '#000000',
        minute: '#000000',
        second: '#000000',
    },
    notchColor: '#000000',
    numbersColor: '#000000',
    showNumbers: true,
    size: 300
}

<AnalogClock {...clockOptions} />
```
